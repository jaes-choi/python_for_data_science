ta_agg['중상'] = ta.groupby('발생지시도')['중상자수'].sum()
ta_agg['대형'] = ta.groupby('발생지시도')['대형사고'].sum()
ta_agg['대형사고율'] = ta.groupby('발생지시도')['대형사고'].mean()
ta_agg['사고건수'] = ta.groupby('발생지시도')['대형사고'].count()
ta_agg

ta_agg.columns
